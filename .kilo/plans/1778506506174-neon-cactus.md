### 2.4 JavaScript — nowe funkcje

#### `initSettingsModal()`
Inicjalizowana w `DOMContentLoaded` razem z innymi modalami.

```js
function initSettingsModal() {
  // Wczytaj zapisany motyw przy starcie
  const saved = localStorage.getItem('propcheck_theme') || 'default';
  applyTheme(saved);
  updateThemeCards(saved);

  // Podepnij otwieranie modalu
  document.querySelectorAll('[data-open-settings]').forEach(el => {
    el.addEventListener('click', () => {
      document.getElementById('settingsModal').classList.add('active');
    });
  });

  // Zamknięcie
  // ...standardowy wzorzec closeBtn, backdrop click, Escape

  // Wybór motywu
  document.querySelectorAll('.theme-card').forEach(card => {
    card.addEventListener('click', () => {
      const theme = card.dataset.theme;
      applyTheme(theme);
      updateThemeCards(theme);
      localStorage.setItem('propcheck_theme', theme);
    });
  });

  // Purge
  document.getElementById('devPurgeBtn').addEventListener('click', () => {
    window.showDeleteConfirmation(
      null,
      'wszystkie dane (mieszkania i usterki)',
      () => {
        localStorage.removeItem('propcheck_properties');
        localStorage.removeItem('propcheck_issues');
        properties = [];
        issues = [];
        renderProperties();
        renderIssuesList();
        updateStats();
        closeSettingsModal();
      },
      'data'
    );
  });

  // Populate
  document.getElementById('devPopulateBtn').addEventListener('click', () => {
    populateDemoData();
    closeSettingsModal();
  });
}

function applyTheme(theme) {
  document.documentElement.setAttribute('data-theme',
    theme === 'default' ? '' : theme
  );
  // lub: theme === 'default' → removeAttribute
}

function updateThemeCards(activeTheme) {
  document.querySelectorAll('.theme-card').forEach(card => {
    card.classList.toggle('active', card.dataset.theme === activeTheme);
  });
}
```

### 2.7 Wczytanie motywu przy starcie (bez flash)

Na początku `<head>` w `dashboard.html` dodajemy inline `<script>` (przed CSS nie można, ale można po linku CSS lub przez DOMContentLoaded z `applyTheme`). Ponieważ jest to pure CSS variables, wystarczy wywołanie w `DOMContentLoaded` — nie będzie widocznego FOUC bo zmieniamy zmienne CSS, nie klasy layout.

### 2.8 Obsługa błędów

- Jeśli localStorage jest niedostępny: try/catch, fallback do motywu domyślnego
- Purge wymaga potwierdzenia (reużycie istniejącego `showDeleteConfirmation`)
- Populate nie pyta o potwierdzenie — szybka akcja dla dewelopera
- Jeśli `populate` wywoływane gdy są dane: czyści i zastępuje (bez ostrzeżenia — kontekst dev-tools)
